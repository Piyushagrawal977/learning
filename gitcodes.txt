
git config --global user.name "name"	  - Enter a user name

git init 				  -  create repository 
git status				  - to see the status
git log				  - to view all the commits
git add "<file name>" 		  -  add the file in staging 
git add --a				  -  add all the file in staging 
git add .     				  - add all the file in staging 
git commit "<file name > " -m "Message "   -  to commit
git commit  -m "Message"  		  - to commit all file from staging 
rm -rf .git 			 	  - to delete the repository 
touch <file name.extenstion>		  - create a file
git clone <Url>			  -taking repository from the network
git clone <Url> "new name "		  -changing the name while cloning the repository from the nework
ls				  - to view the list of files in the directory
pwd                                             		  - to see the present directory
cd				  - to change the directory
touch .gitignore			  - creating a file for ignoring the directory and files
git diff 				  - comparing the working directory to staging area
git diff --staged			  - comparing the working directory to previous commit
git restore  --staged 	<file name> 	  - to restor staging file
